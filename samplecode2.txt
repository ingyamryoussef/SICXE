COPY START 0
    EXTDEF BUFFER,BUFEND,LENGTH
    EXTREF RDREC,WRREC
FIRST STL RETADR
CLOOP +JSUB RDREC
    LDA LENGTH
    COMP #0
    JEQ ENDFIL
    +JSUB WRREC
    J CLOOP
ENDFIL BYTE C'EOF'
    STA BUFFER
    LDA #3
    STA LENGTH
    +JSUB WRREC
    J @RETADR
RETADR RESW 1
LENGTH RESW 1
    BYTE C'EOF'
BUFFER RESB 4096
BUFEND EQU *
MAXLEN EQU BUFFEND-BUFFER
RDREC CSET NULL
EXTREF BUFFER,LENGTH,BUFEND
    CLEAR X
    CLEAR A
    CLEAR S
    LDT MAXLEN
RLOOP TD INPUT
    JEQ RLOOP
    RD INPUT
    COMPR A,S
    JEQ EXIT
    +STCH BUFFER,X
    TIXR T
    JLT RLOOP
EXIT +STX LENGTH
    RSUB NULL
INPUT BYTE X'F1'
